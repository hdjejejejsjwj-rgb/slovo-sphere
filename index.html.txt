  <div class="feed" id="feed"></div>

  <script>
    const tg = window.Telegram.WebApp;
    tg.ready();
    tg.expand();

    // Тестовые слова (можешь менять прямо тут)
    const words = [
      { id: 1, word: 'мир', likes: 15, money: 320 },
      { id: 2, word: 'peace', likes: 8, money: 120 },
      { id: 3, word: '世界', likes: 5, money: 70 },
      { id: 4, word: 'a', likes: 3, money: 40 },
      { id: 5, word: 'любовь', likes: 22, money: 510 }
    ];

    const feed = document.getElementById('feed');
    words.forEach(w => {
      const card = document.createElement('div');
      card.className = 'card';
      card.innerHTML = `
        <div class="header">
          <div class="rank">№${w.id}</div>
          <div class="star">★ ${w.likes}</div>
        </div>
        <div class="word">${w.word}</div>
        <div class="triangle">▲ ${w.money} ₽</div>
        <button onclick="support(${w.id}, this)">Поддержать</button>
      `;
      feed.appendChild(card);
    });

    function support(id, btn) {
      btn.textContent = 'Спасибо!';
      btn.style.background = '#aaa';
      setTimeout(() => {
        btn.textContent = 'Поддержать';
        btn.style.background = '#2ecc71';
      }, 2000);
    }
  </script>
